<script>


const canvas = document.getElementById("game");

const ctx = canvas.getContext("2d");


let nainy = {x:50,y:50};


let rivals = [

{name:"Girl1",x:300,y:50},

{name:"Girl2",x:300,y:200},

{name:"Girl3",x:50,y:300}

];


let sara = {name:"Sara",x:200,y:300};


let shubman = {x:350,y:350};


function drawCharacter(x,y,color,label){


ctx.fillStyle=color;

ctx.beginPath();

ctx.arc(x,y,15,0,Math.PI*2);

ctx.fill();


ctx.fillStyle="white";

ctx.font="12px Arial";

ctx.fillText(label,x-20,y-20);


}


function draw(){


ctx.clearRect(0,0,400,400);


drawCharacter(nainy.x,nainy.y,"yellow","Nainy");


rivals.forEach(r=>{

drawCharacter(r.x,r.y,"pink",r.name);

});


drawCharacter(sara.x,sara.y,"red","Sara");


drawCharacter(shubman.x,shubman.y,"green","Shubman Gill");


}


function move(dir){


if(dir==="up") nainy.y-=20;

if(dir==="down") nainy.y+=20;

if(dir==="left") nainy.x-=20;

if(dir==="right") nainy.x+=20;


checkCollisions();

draw();


}


function distance(a,b){


return Math.hypot(a.x-b.x,a.y-b.y);


}


function checkCollisions(){


for(let r of rivals){


if(distance(nainy,r)<20){


document.getElementById("msg").innerText="Nainy lost! Rival girl blocked her.";

reset();


}


}


if(distance(nainy,sara)<20){


document.getElementById("msg").innerText="Sara defeated Nainy!";

reset();


}


if(distance(nainy,shubman)<20){


document.getElementById("msg").innerText="❤️ nainy ka dil. Shubman gill ❤️";


}


}


function reset(){


nainy.x=50;

nainy.y=50;


}


draw();


</script>
